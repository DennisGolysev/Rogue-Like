public class RoomDirection {
    private int value;

    public RoomDirection(int value) {
        this.value = value;
    }

    public void inc(int num) {
        for(int i = 0; i < num; i++) {
            if(value < 3) {
                value = value + 1;
            } else {
                value = 0;
            }
        }
    }
}

public class RoomSource {
    private int side_;
	private Room source_;

    public RoomSource(int side, Room source) {
        this.side_ = side;
		this.source_ = source;
    }

    public int GetSide() {
        return side_;
    }
	
	public Room GetSource() {
		return source_;
	}
}
